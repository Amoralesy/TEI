   att.global.rendition.attributes,
   att.global.linking.attributes,

   att.combinable.attributes,
   att.deprecated.attributes,
   (
      ( model.glossLike | model.descLike )*,
      ( content | valList )*,
      constraintSpec*,
      exemplum*,
      remarks*,
      listRef*
   )
}
